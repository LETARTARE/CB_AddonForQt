607  wxString ext = _file.AfterLast('.') ;
608 //1- file *.ui (forms)
609  if ( ext.Matches(m_UI) )
610  fout += m_UI + "_" + name + DOT_EXT_H ;
611  else
612 //2- file *.qrc (resource)
613  if (ext.Matches(m_Qrc) )
614  fout += m_Qrc + "_" + name + DOT_EXT_CPP ;
615  else
616 //3- file *.h or *hpp with 'Q_OBJECT' and/or 'Q_GADGET'
617  if (ext.Matches(EXT_H) || ext.Matches(EXT_HPP) )
618  fout += m_Moc + "_" + name + DOT_EXT_CPP ;
619  else
620 //4- file *.cpp with 'Q_OBJECT' and/or 'Q_GADGET'
621  if (ext.Matches(EXT_CPP) )
622  fout += name + DOT_EXT_MOC ;
623 
624  fout = fout.AfterLast(Slash) ;
625 
626  return fout ;
